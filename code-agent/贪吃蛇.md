
# 贪吃蛇游戏实现

基于您的需求文档，我设计了一个完整的贪吃蛇游戏。下面是实现代码：

```html
<!DOCTYPE html>
<html lang="zh-CN">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>贪吃蛇游戏</title>
    <style>
        body {
            font-family: 'Arial', sans-serif;
            display: flex;
            flex-direction: column;
            align-items: center;
            justify-content: center;
            height: 100vh;
            margin: 0;
            background-color: #f0f0f0;
        }
        
        .game-container {
            display: flex;
            flex-direction: column;
            align-items: center;
            gap: 20px;
        }
        
        .score-board {
            font-size: 24px;
            font-weight: bold;
            color: #333;
        }
        
        #game-canvas {
            border: 2px solid #333;
            background-color: #fff;
            box-shadow: 0 0 10px rgba(0, 0, 0, 0.2);
        }
        
        .controls {
            display: flex;
            gap: 10px;
        }
        
        button {
            padding: 10px 20px;
            font-size: 16px;
            background-color: #4CAF50;
            color: white;
            border: none;
            border-radius: 4px;
            cursor: pointer;
            transition: background-color 0.3s;
        }
        
        button:hover {
            background-color: #45a049;
        }
        
        button:disabled {
            background-color: #cccccc;
            cursor: not-allowed;
        }
        
        .game-status {
            font-size: 18px;
            margin-top: 10px;
            height: 24px;
            color: #d9534f;
            font-weight: bold;
        }
    </style>
</head>
<body>
    <div class="game-container">
        <div class="score-board">分数: <span id="score">0</span></div>
        <canvas id="game-canvas" width="400" height="400"></canvas>
        <div class="controls">
            <button id="start-btn">开始游戏</button>
            <button id="restart-btn" disabled>重新开始</button>
        </div>
        <div class="game-status" id="game-status"></div>
    </div>

    <script>
        // 游戏配置
        const GRID_SIZE = 20;
        const GRID_WIDTH = 20;
        const GRID_HEIGHT = 20;
        const GAME_SPEED = 150; // 毫秒

        // 游戏元素
        const canvas = document.getElementById('game-canvas');
        const ctx = canvas.getContext('2d');
        const scoreElement = document.getElementById('score');
        const startButton = document.getElementById('start-btn');
        const restartButton = document.getElementById('restart-btn');
        const gameStatusElement = document.getElementById('game-status');

        // 游戏状态
        let snake = [];
        let food = {};
        let direction = 'right';
        let nextDirection = 'right';
        let score = 0;
        let gameInterval = null;
        let gameRunning = false;

        // 初始化游戏
        function initGame() {
            // 初始化蛇
            snake = [
                {x: 10, y: 10},
                {x: 9, y: 10},
                {x: 8, y: 10}
            ];
            
            // 生成食物
            generateFood();
            
            // 重置分数和方向
            score = 0;
            scoreElement.textContent = score;
            direction = 'right';
            nextDirection = 'right';
            
            // 清除游戏状态
            gameStatusElement.textContent = '';
            
            // 绘制初始状态
            draw();
        }

        // 生成食物
        function generateFood() {
            // 创建所有可能的位置
            const possiblePositions = [];
            for (let x = 0; x < GRID_WIDTH; x++) {
                for (let y = 0; y < GRID_HEIGHT; y++) {
                    possiblePositions.push({x, y});
                }
            }
            
            // 移除蛇身占据的位置
            const availablePositions = possiblePositions.filter(position => {
                return !snake.some(segment => segment.x === position.x && segment.y === position.y);
            });
            
            // 随机选择一个可用位置
            if (availablePositions.length > 0) {
                const randomIndex = Math.floor(Math.random() * availablePositions.length);
                food = availablePositions[randomIndex];
            }
        }

        // 游戏主循环
        function gameLoop() {
            // 更新方向
            direction = nextDirection;
            
            // 移动蛇
            const head = {...snake[0]};
            
            switch (direction) {
                case 'up':
                    head.y -= 1;
                    break;
                case 'down':
                    head.y += 1;
                    break;
                case 'left':
                    head.x -= 1;
                    break;
                case 'right':
                    head.x += 1;
                    break;
            }
            
            // 检查碰撞
            if (checkCollision(head)) {
                gameOver();
                return;
            }
            
            // 将新头部添加到蛇身
            snake.unshift(head);
            
            // 检查是否吃到食物
            if (head.x === food.x && head.y === food.y) {
                // 增加分数
                score += 1;
                scoreElement.textContent = score;
                
                // 生成新食物
                generateFood();
            } else {
                // 如果没吃到食物，移除尾部
                snake.pop();
            }
            
            // 绘制游戏
            draw();
        }

        // 检查碰撞
        function checkCollision(position) {
            // 检查墙壁碰撞
            if (
                position.x < 0 || 
                position.x >= GRID_WIDTH || 
                position.y < 0 || 
                position.y >= GRID_HEIGHT
            ) {
                return true;
            }
            
            // 检查自身碰撞（跳过头部）
            for (let i = 1; i < snake.length; i++) {
                if (snake[i].x === position.x && snake[i].y === position.y) {
                    return true;
                }
            }
            
            return false;
        }

        // 绘制游戏
        function draw() {
            // 清空画布
            ctx.clearRect(0, 0, canvas.width, canvas.height);
            
            // 绘制网格线
            ctx.strokeStyle = '#e0e0e0';
            for (let i = 0; i <= GRID_WIDTH; i++) {
                ctx.beginPath();
                ctx.moveTo(i * GRID_SIZE, 0);
                ctx.lineTo(i * GRID_SIZE, canvas.height);
                ctx.stroke();
            }
            for (let i = 0; i <= GRID_HEIGHT; i++) {
                ctx.beginPath();
                ctx.moveTo(0, i * GRID_SIZE);
                ctx.lineTo(canvas.width, i * GRID_SIZE);
                ctx.stroke();
            }
            
            // 绘制蛇
            snake.forEach((segment, index) => {
                ctx.fillStyle = index === 0 ? '#333' : '#4CAF50'; // 头部和身体不同颜色
                ctx.fillRect(
                    segment.x * GRID_SIZE, 
                    segment.y * GRID_SIZE, 
                    GRID_SIZE, 
                    GRID_SIZE
                );
                
                // 绘制边框
                ctx.strokeStyle = '#fff';
                ctx.strokeRect(
                    segment.x * GRID_SIZE, 
                    segment.y * GRID_SIZE, 
                    GRID_SIZE, 
                    GRID_SIZE
                );
            });
            
            // 绘制食物
            ctx.fillStyle = '#d9534f';
            ctx.beginPath();
            ctx.arc(
                food.x * GRID_SIZE + GRID_SIZE / 2,
                food.y * GRID_SIZE + GRID_SIZE / 2,
                GRID_SIZE / 2 - 2,
                0,
                Math.PI * 2
            );
            ctx.fill();
        }

        // 游戏结束
        function gameOver() {
            clearInterval(gameInterval);
            gameRunning = false;
            gameStatusElement.textContent = `游戏结束！最终得分: ${score}`;
            startButton.disabled = true;
            restartButton.disabled = false;
        }

        // 开始游戏
        function startGame() {
            if (!gameRunning) {
                initGame();
                gameInterval = setInterval(gameLoop, GAME_SPEED);
                gameRunning = true;
                startButton.disabled = true;
                restartButton.disabled = true;
            }
        }

        // 重新开始游戏
        function restartGame() {
            if (gameInterval) {
                clearInterval(gameInterval);
            }
            startGame();
        }

        // 键盘控制
        document.addEventListener('keydown', (event) => {
            if (!gameRunning) return;
            
            switch (event.key) {
                case 'ArrowUp':
                    if (direction !== 'down') nextDirection = 'up';
                    break;
                case 'ArrowDown':
                    if (direction !== 'up') nextDirection = 'down';
                    break;
                case 'ArrowLeft':
                    if (direction !== 'right') nextDirection = 'left';
                    break;
                case 'ArrowRight':
                    if (direction !== 'left') nextDirection = 'right';
                    break;
            }
        });

        // 按钮事件
        startButton.addEventListener('click', startGame);
        restartButton.addEventListener('click', restartGame);

        // 初始化界面
        initGame();
    </script>
</body>
</html>
```

## 功能说明

这个贪吃蛇游戏实现了需求文档中的所有功能：

1. **核心功能**：
   - 使用键盘方向键控制蛇的移动
   - 随机生成食物（红色圆形）
   - 碰撞检测（墙壁和自身）
   - 分数系统（每吃一个食物得1分）

2. **游戏界面**：
   - 20×20网格地图，带有网格线
   - 实时显示当前分数
   - 游戏状态提示
   - 开始和重新开始按钮

3. **游戏逻辑**：
   - 蛇初始长度为3格
   - 固定移动速度
   - 食物被吃掉后立即在新的随机位置生成
   - 蛇身随着吃食物而增长

4. **用户体验**：
   - 简洁明了的界面设计
   - 响应灵敏的控制
   - 清晰的游戏状态反馈
   - 即时的分数更新

5. **异常处理**：
   - 防止蛇反向移动（例如向右移动时不能立即向左）
   - 处理游戏边界情况
   - 确保食物不会生成在蛇身上

游戏使用Canvas实现渲染，性能良好，兼容主流浏览器。您可以直接复制这段代码到HTML文件中运行。
